// Database configuration
define('DB_HOST', 'localhost');        // Database server (127.0.0.1 or localhost)
define('DB_NAME', 'youngtur_chaitanya_resort'); // Database name
define('DB_USER', 'youngtur_admin_vikram');             // Your MySQL username (default is 'root' for local)
define('DB_PASS', 'Password@2025');                 // Your MySQL password (leave empty if no password)
define('DB_PORT','3306');
define('DB_CHARSET', 'utf8mb4');

/**
 * Get database connection
 * @return PDO|null Returns PDO connection or null on failure
 */
function getDBConnection() {
    try {
        $dsn = "mysql:host=" . DB_HOST . ";dbname=" . DB_NAME . ";port=DB_PORT;charset=" . DB_CHARSET;
        $options = [
            PDO::ATTR_ERRMODE            => PDO::ERRMODE_EXCEPTION,
            PDO::ATTR_DEFAULT_FETCH_MODE => PDO::FETCH_ASSOC,
            PDO::ATTR_EMULATE_PREPARES   => false,
        ];
        
        $pdo = new PDO($dsn, DB_USER, DB_PASS, $options);
        return $pdo;
    } catch (PDOException $e) {
        error_log("Database connection error: " . $e->getMessage());
        return null;
    }
}
